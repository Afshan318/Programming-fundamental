 Find Largest Number.

#include <iostream>
using namespace std;

int main() {
    int num, largest = 0;

    cout << "Enter a number: ";
    cin >> num;

    while (num > 0) {
        int digit = num % 10;
        if (digit > largest) {
            largest = digit;
        }
        num /= 10;
    }

    cout << "The largest digit is: " << largest;

    return 0;
}
