Pyramid shape with stars.


#include <iostream>
using namespace std;

int main() {
    int rows = 5;  // Number of rows for the pyramid
    
    for (int i = 1; i <= rows; i++) {
        // Printing spaces to center the stars
        for (int j = 1; j <= rows - i; j++) {
            cout << " ";
        }
        
        // Printing stars
        for (int k = 1; k <= (2 * i - 1); k++) {
            cout << "*";
        }
        
        cout << endl;
    }
    
    return 0;
}
