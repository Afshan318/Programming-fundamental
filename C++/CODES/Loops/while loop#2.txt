Sum of Natural Number up to N.

#include <iostream>
using namespace std;

int main() {
    int N, sum = 0;

    cout << "Enter a number: ";
    cin >> N;

    int i = 1;
    while (i <= N) {
        sum += i;
        i++;
    }

    cout << "Sum of natural numbers up to " << N << " is: " << sum;

    return 0;
}
