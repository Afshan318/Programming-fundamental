Print Factorial Series.



#include <iostream>
using namespace std;

int factorial(int num) {
    int fact = 1;
    for (int i = 1; i <= num; i++) {
        fact *= i;
    }
    return fact;
}

int main() {
    int n = 5;  // Number of terms
    for (int i = 1; i <= n; i++) {
        cout << "Factorial of " << i << " is: " << factorial(i) << endl;
    }
    return 0;
}
