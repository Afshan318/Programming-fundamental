Harmonic Series.



#include <iostream>
using namespace std;

int main() {
    int n = 10;  // Number of terms
    double sum = 0.0;
    for (int i = 1; i <= n; i++) {
        sum += 1.0 / i;
    }
    cout << "Sum of the first " << n << " terms of the harmonic series is: " << sum << endl;
    return 0;
}
