Average of N Number.

#include <iostream>
using namespace std;

int main() {
    int n, number, sum = 0;
    cout << "How many numbers do you want to average? ";
    cin >> n;

    int i = 0;
    do {
        cout << "Enter a number: ";
        cin >> number;
        sum += number;
        i++;
    } 
while (i < n);

    cout << "The average is " << (sum / n) << endl;
    return 0;
}
