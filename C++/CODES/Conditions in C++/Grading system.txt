#include <iostream>
using namespace std;

int main() {
    int marks;
    cout << "Enter marks (0 to 100): ";
    cin >> marks;

    if (marks >= 90 && marks <= 100) {
        cout << "Grade A" << endl;
    } 
else if (marks >= 75) {
        cout << "Grade B" << endl;
    } 
else if (marks >= 50) {
        cout << "Grade C" << endl;
    }
 else if (marks >= 0) {
        cout << "Grade F" << endl;
    } 
else {
        cout << "Invalid marks" << endl;
    }

    return 0;
}
